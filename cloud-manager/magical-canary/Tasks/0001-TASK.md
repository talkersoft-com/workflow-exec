### Service implementation

Add this method as a sibling to `ListByAssignmentAsync` (around line 105). Pattern:

```csharp
public async Task<DtoModel.PagedResult<DtoModel.PlaybookRun>> ListAsync(
    string? vmPublicId,
    string? playbookPublicId,
    PlaybookRunStatus? status,
    int page,
    int pageSize)
{
    page     = Math.Max(1, page);
    pageSize = Math.Clamp(pageSize, 1, 200);

    var q = _context.PlaybookRuns.AsNoTracking();

    if (!string.IsNullOrWhiteSpace(playbookPublicId))
    {
        var pbGuid = await _context.Playbooks.GuidByPublicIdAsync(playbookPublicId);
        if (pbGuid is null)
            return new DtoModel.PagedResult<DtoModel.PlaybookRun>(Array.Empty<DtoModel.PlaybookRun>(), page, pageSize, 0);
        q = q.Where(r => r.PlaybookId == pbGuid.Value);
    }

    if (status.HasValue)
        q = q.Where(r => r.Status == status.Value);

    if (!string.IsNullOrWhiteSpace(vmPublicId))
    {
        var vmGuid = await _context.VirtualMachines
            .Where(v => v.PublicId == vmPublicId)
            .Select(v => (Guid?)v.Id).FirstOrDefaultAsync();
        if (vmGuid is null)
            return new DtoModel.PagedResult<DtoModel.PlaybookRun>(Array.Empty<DtoModel.PlaybookRun>(), page, pageSize, 0);
        q = q.Where(r => _context.PlaybookRunTargets.Any(t => t.RunId == r.Id && t.VmId == vmGuid.Value));
    }

    var total = await q.CountAsync();
    var rows  = await q.OrderByDescending(r => r.StartedAt)
                       .Skip((page - 1) * pageSize)
                       .Take(pageSize)
                       .ToListAsync();

    // Batch playbook public_id resolution — never per-row.
    var pbIds = rows.Select(r => r.PlaybookId).Distinct().ToList();
    var pbMap = await _context.Playbooks.AsNoTracking()
        .Where(p => pbIds.Contains(p.Id))
        .ToDictionaryAsync(p => p.Id, p => p.PublicId);

    var items = rows.Select(r =>
    {
        var dto = _mapper.Map<DtoModel.PlaybookRun>(r);
        dto.PlaybookId    = pbMap.TryGetValue(r.PlaybookId, out var pid) ? pid : r.PlaybookId.ToString();
        dto.VaultRunToken = null; // scrub — only GetByIdAsync surfaces the token
        return dto;
    }).ToList();

    return new DtoModel.PagedResult<DtoModel.PlaybookRun>(items, page, pageSize, total);
}
```

## Notes
- Helper `GuidByPublicIdAsync` already exists on every `DbSet` extension — check `Entities/PublicId/` if signature surprises you.
- `VmPublicId == null` lookup intentionally returns empty `PagedResult` with `total = 0` rather than throwing — clients should not crash on a stale-bookmarked filter.
- Do not change `ListByAssignmentAsync`. Both methods coexist.
